    # 6.6 获胜判断

    最后就是来完善游戏的规则了。网页五子棋的规则很简单，就两条：不能重复落子、连成五子就算赢。

    然后在鼠标单击操作中定义两个变量mtxPosX和mtxPosY，用来保存棋子在数组中的位置。这两个变量的赋值放在棋子坐标的赋值中，如下。

    var mtxPosX, mtxPosY;
for (var x = 0; x < 19; x++) {
    if (Math.abs(event.offsetX - (10 + x*20)) < 10) {
        arcPosX = 10 + x*20;
        mtxPosX = x;
    }
    if (Math.abs(event.offsetY - (10 + x*20)) < 10) {
        arcPosY = 10 + x*20;
        mtxPosY = x;
    }
}

    同时，落子方面也要进行一些修改。首先就是判断落子的位置是不是空的，即matrix[mtxPosX][mtxPosY]是不是等于0，如果等于0，才能运行响应的绘制棋子的操作。绘制完成后，要根据落子的颜色设定matrix[mtxPosX][mtxPosY]的值，黑子（蓝色）为1，白子为2.修改后代码如下。

    if (matrix[mtxPosX][mtxPosY] === 0) {
    blorwh = !blorwh;
    ctx.beginPath();
    ctx.arc(arcPosX, arcPosY, 10, 0, Math.PI*2, false);
    if (blorwh) {
        ctx.fillStyle = "blue";
        matrix[mtxPosX][mtxPosY] = 1;
    } else {
        ctx.fillStyle = "white";
        ctx.stroke();
        matrix[mtxPosX][mtxPosY] = 2;
    }
    ctx.fill();
}

    完成后再测试一下，没有什么问题的话，就可以来实现第二条规则了。第二条规则比较麻烦，需要判断落子各个方向上是不是有5个子相连。我们先选择水平方向进行检测，从（mtxPosX, mtxPosY）这个位置出发，左边的位置就是（mtxPosX - 1, mtxPosY），如果（mtxPosX - 1, mtxPosY）的位置上是和（mtxPosX, mtxPosY）位置向的状态（落子）一样的话，就接着判断位置（mtxPosX - 2, mtxPosY），否则就从位置（mtxPosX - 1, mtxPosY）向右判断5个子是不是都是一样的。以此类推。完成后代码如下。

    if (matrix[mtxPosX - 1][mtxPosY] === matrix[mtxPosX][mtxPosY]) {
    if (matrix[mtxPosX - 2][mtxPosY] === matrix[mtxPosX][mtxPosY]) {
        if (matrix[mtxPosX - 3][mtxPosY] === matrix[mtxPosX][mtxPosY]) {
            if (matrix[mtxPosX - 4][mtxPosY] === matrix[mtxPosX][mtxPosY]) {
                winFlag = 1;
            } else {
                if (matrix[mtxPosX + 1][mtxPosY] !== matrix[mtxPosX][mtxPosY]) {
                    winFlag = 0;
                } else {
                    winFlag = 1;
                }
            }
        } else {
            for (var w = 0; w < 2; w++) {
                if (matrix[mtxPosX + w + 1][mtxPosY] !== matrix[mtxPosX][mtxPosY]) {
                    winFlag = 0;
                    break;
                } else {
                    winFlag = 1;
                }
            }
        }
    } else {
        for (var w = 0; w < 3; w++) {
            if (matrix[mtxPosX + w + 1][mtxPosY] !== matrix[mtxPosX][mtxPosY]) {
                winFlag = 0;
                break;
            } else {
                winFlag = 1;
            }
        }
    }
} else {
    for (var w = 0; w < 4; w++) {
        if (matrix[mtxPosX + w + 1][mtxPosY] !== matrix[mtxPosX][mtxPosY]) {
            winFlag = 0;
            break;
        } else {
            winFlag = 1;
        }
    }
}

    if (winFlag === 1) {
